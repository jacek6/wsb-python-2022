        self.x = x
        self.y = y
        self.krok_x = 2
        self.krok_y = 3

        self.x += self.krok_x
        if self.x < 0:
            self.krok_x = -self.krok_x

        self.y += self.krok_y
        if self.y < 0:
            self.krok_y = -self.krok_y
            self.y = 0
        elif self.y + 50 > WYSOKOSC_OKNA:
            self.krok_y = -self.krok_y
            self.y = WYSOKOSC_OKNA - 50
